class ModalDialog extends HTMLElement {
  constructor() {
    super();
    this.querySelector('[data-close]')?.addEventListener('click', this.close.bind(this));
  }
  
  open() {
    this.setAttribute('open', '');
    this.setAttribute('aria-modal', 'true');
    this.trapFocus();
    document.body.classList.add('overflow-hidden');
  }
  
  close() {
    this.removeAttribute('open');
    this.removeAttribute('aria-modal');
    this.removeTrapFocus();
    document.body.classList.remove('overflow-hidden');
  }
  
  trapFocus() {
    const focusableElements = this.querySelectorAll(
      'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])'
    );
    this.firstFocusable = focusableElements[0];
    this.lastFocusable = focusableElements[focusableElements.length - 1];
    this.addEventListener('keydown', this.handleKeydown);
    this.firstFocusable?.focus();
  }
  
  handleKeydown = (e) => {
    if (e.key === 'Escape') { this.close(); return; }
    if (e.key === 'Tab') {
      if (e.shiftKey && document.activeElement === this.firstFocusable) {
        e.preventDefault(); this.lastFocusable?.focus();
      } else if (!e.shiftKey && document.activeElement === this.lastFocusable) {
        e.preventDefault(); this.firstFocusable?.focus();
      }
    }
  }
  
  removeTrapFocus() { this.removeEventListener('keydown', this.handleKeydown); }
}
customElements.define('modal-dialog', ModalDialog);

10.1 Cart Drawer (Section Rendering)
â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
class CartDrawer extends HTMLElement {
  constructor() {
    super();
    document.addEventListener('cart:refresh', this.refreshCart.bind(this));
    this.addEventListener('click', this.onClick.bind(this));
  }
  
  async onClick(e) {
    if (e.target.matches('[data-remove-from-cart]')) {
      e.preventDefault();
      await this.removeFromCart(e.target.dataset.lineIndex);
    }
  }
  
  async removeFromCart(lineIndex) {
    const url = window.routes.cart_change_url || '/cart/change.js';
    try {
      await fetch(url, {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({ line: lineIndex, quantity: 0 })
      });
      await this.refreshCart();
    } catch (err) { console.error(err); }
  }
  
  async refreshCart() {
    const cartUrl = window.routes.cart_url || '/cart';
    const res = await fetch(`${cartUrl}?section_id=cart-drawer`);
    const text = await res.text();
    
    const html = new DOMParser().parseFromString(text, 'text/html');
    
    this.renderElement(html, '[data-cart-items]');
    this.renderElement(html, '[data-cart-footer]');
    this.updateCartCount(html.querySelector('[data-cart-count]')?.textContent);
    this.open();
  }

  renderElement(sourceDoc, selector) {
    const source = sourceDoc.querySelector(selector);
    const target = this.querySelector(selector);
    if (source && target) target.innerHTML = source.innerHTML;
  }
  
  updateCartCount(count) {
    if (count) document.querySelectorAll('[data-cart-count]').forEach(el => el.textContent = count);
  }
  
  open() { this.setAttribute('open', ''); this.trapFocus(); }
  close() { this.removeAttribute('open'); }
}
customElements.define('cart-drawer', CartDrawer);

class StickyAtc extends HTMLElement {
  constructor() {
    super();
    this.mainButton = document.querySelector('.product-form__submit');

        if (!entry.isIntersecting) {
          this.bar.classList.add('is-visible');

    }, { threshold: 0 });
    
    observer.observe(this.mainButton);
  }
}

11.1 Predictive Search
â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
class PredictiveSearch extends HTMLElement {
  constructor() {
    super();
    this.input = this.querySelector('input[type="search"]');
    this.results = this.querySelector('[data-predictive-search]');
    this.input.addEventListener('input', this.debounce(this.onChange.bind(this), 300));
    this.abortController = new AbortController();
  }
  
  onChange() {
    const term = this.input.value.trim();
    if (term.length < 3) { this.close(); return; }
    
    this.abortController.abort();
    this.abortController = new AbortController();
    
    const url = `${window.routes.predictive_search_url}?q=${encodeURIComponent(term)}&resources[type]=product&section_id=predictive-search`;
    
    fetch(url, { signal: this.abortController.signal })
      .then(res => res.text())
      .then(text => {
        const html = new DOMParser().parseFromString(text, 'text/html');
        this.results.innerHTML = html.querySelector('#shopify-section-predictive-search').innerHTML;
        this.setAttribute('open', 'true');
      })
      .catch(err => { if (err.name !== 'AbortError') console.error(err); });
  }
  
  debounce(fn, wait) {
    let t;
    return (...args) => { clearTimeout(t); t = setTimeout(() => fn.apply(this, args), wait); };
  }
  
  close() { this.removeAttribute('open'); }
}
customElements.define('predictive-search', PredictiveSearch);

  constructor() {
    super();
    this.addEventListener('change', this.onChange.bind(this));
  }

  render(url) {
    history.pushState({}, '', url);

      const doc = new DOMParser().parseFromString(html, 'text/html');
      document.querySelector('#product-grid').innerHTML = doc.querySelector('#product-grid').innerHTML;
    });
  }
}
